14/// Searches a single file and returns all lines that match the query.
15///
16/// The file is read line-by-line using a [`BufReader`] to avoid loading
17/// the entire file into memory at once. Each line is tested according to
18/// the active flags in `params`:
19///
20/// | Flag          | Behavior                                                    |
21/// |---------------|--------------------------------------------------------------|
22/// | `ignore_case` | Both the query and the line are lowercased before comparison.|
23/// | `whole_word`  | Delegates to [`is_whole_word_match`].                        |
24/// | *(neither)*   | Plain substring search via [`str::contains`].                |
25///
26/// Lines that produce an I/O error during reading are silently skipped.
27///
28/// # Arguments
29///
30/// * `params` - Search configuration (query string, flags, etc.).
31/// * `path`   - Filesystem path of the file to search.
32///
33/// # Returns
34///
35/// A `Vec<LineMatchModel>` containing every matching line with its 1-based
36/// line number, or an [`Error`] if the file cannot be opened.
37pub fn search_file(params: &Parameters, path: &str) -> Result<Vec<LineMatchModel>, Error> {
38    // Normalize the query once outside the per-line loop.
39    let query: Cow<str> = if params.ignore_case {
40        Cow::Owned(params.query.to_lowercase())
41    } else {
42        Cow::Borrowed(&params.query)
43    };
44
45    let file = File::open(path)?;
46    let reader = BufReader::new(file);
47
48    let results = reader
49        .lines()
50        .enumerate()
51        .filter_map(|(i, line_result)| {
52            // Skip lines that fail to decode (e.g. invalid UTF-8).
53            let line = line_result.ok()?;
54
55            let normalized_line = if params.ignore_case {
56                line.to_lowercase()
57            } else {
58                line.clone()
59            };
60
61            let matched = if params.whole_word {
62                is_whole_word_match(&normalized_line, &query)
63            } else {
64                normalized_line.contains(query.as_ref())
65            };
66
67            if matched {
68                // Store the original (non-lowercased) line for display.
69                Some(LineMatchModel::new(i + 1, line.to_string()))
70            } else {
71                None
72            }
73        })
74        .collect();
75
76    Ok(results)
77}

79/// Recursively searches all `.txt` files under `dir` in parallel and returns
80/// per-file results sorted by file name.
81///
82/// The directory tree is walked with [`WalkDir`], then all eligible files are
83/// processed concurrently using a Rayon parallel iterator. Files that cannot
84/// be read are skipped with an error message printed to `stderr`. Files with
85/// no matches are excluded from the returned vector.
86///
87/// # Arguments
88///
89/// * `dir`    - Root directory to walk (recursive).
90/// * `params` - Search configuration forwarded to [`search_file`].
91///
92/// # Returns
93///
94/// A `Vec<FileMatchModel>` — one entry per file that contained at least one
95/// match, sorted alphabetically by file path.
96pub fn search_directory(dir: &PathBuf, params: &Parameters) -> Vec<FileMatchModel> {
97    // Collect eligible files upfront so Rayon can distribute them across threads.
98    let entries: Vec<_> = WalkDir::new(dir)
99        .into_iter()
100        .filter_map(|e| e.ok())
101        .filter(|e| {
102            e.path().is_file() && e.path().extension().and_then(|s| s.to_str()) == Some("txt")
103        })
104        .collect();
105
106    let mut result: Vec<FileMatchModel> = entries
107        .into_par_iter()
108        .filter_map(|entry| {
109            let path = entry.path();
110            let path_str = path.to_string_lossy().to_string();
111
112            match search_file(params, &path_str) {
113                Ok(lines) if !lines.is_empty() => Some(FileMatchModel::new(path_str, lines)),
114                Ok(_) => None, // File opened fine but had no matches.
115                Err(e) => {
116                    eprintln!("Couldn't read {:?}: {}", path, e);
117                    None
118                }
119            }
120        })
121        .collect();
122
123    // Parallel iteration produces results in non-deterministic order; sort for
124    // consistent, predictable output.
125    result.sort_unstable_by(|a, b| a.file_name.cmp(&b.file_name));
126    result
127}

129/// Returns `true` when `query` appears in `line` as a whole word.
130///
131/// A *whole-word* match requires that the characters immediately before and
132/// after the occurrence are either absent (start/end of string) or are not
133/// alphanumeric and not underscores — mirroring the word-boundary semantics
134/// of most grep utilities.
135///
136/// # Arguments
137///
138/// * `line`  - The (possibly normalized) line to search.
139/// * `query` - The (possibly normalized) search term.
140///
141/// # Returns
142///
143/// `false` when `query` is empty (avoids a match at every position).
144pub fn is_whole_word_match(line: &str, query: &str) -> bool {
145    if query.is_empty() {
146        return false;
147    }
148
149    line.match_indices(query).any(|(start, _)| {
150        let end = start + query.len();
151
152        // The character *before* the match must be a word boundary (or absent).
153        let before_ok = line[..start]
154            .chars()
155            .next_back()
156            .map_or(true, |c| !c.is_alphanumeric() && c != '_');
157
158        // The character *after* the match must be a word boundary (or absent).
159        let after_ok = line[end..]
160            .chars()
161            .next()
162            .map_or(true, |c| !c.is_alphanumeric() && c != '_');
163
164        before_ok && after_ok
165    })
166}
