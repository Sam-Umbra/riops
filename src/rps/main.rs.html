18/// Executes the search based on the provided [`Parameters`] and prints results to stdout.
19///
20/// Dispatches to either single-file or directory mode depending on whether
21/// `params.directory` is set. Two output modes are supported, controlled by
22/// `params.simple_search`:
23///
24/// | Mode       | Output                                                          |
25/// |------------|-----------------------------------------------------------------|
26/// | Normal     | Every matching line prefixed with its 1-based line number.      |
27/// | Simple     | One summary line per file: `<query> in <file>: N occurrences(s)` |
28///
29/// # Errors
30///
31/// Returns an error if:
32/// * `--directory` is not set and no `--file-path` is provided.
33/// * The specified file cannot be opened or read.
34fn run(params: Parameters) -> Result<(), Box<dyn Error>> {
35    if let Some(dir) = &params.directory {
36        // --- Directory mode (parallel) ---
37        let file_matches = search_directory(dir, &params);
38
39        if params.simple_search {
40            // One summary line per matched file.
41            for file_match in &file_matches {
42                println!(
43                    "{} in {}: {} occurrences(s)",
44                    params.query,
45                    file_match.file_name,
46                    file_match.lines.len()
47                )
48            }
49        } else {
50            // Full output: file header followed by every matching line.
51            for file_match in file_matches {
52                println!("{file_match}\n");
53            }
54        }
55    } else {
56        // --- Single-file mode ---
57        let path = params
58            .file_path
59            .as_deref()
60            .ok_or("File path is required when not using --directory")?;
61
62        let line_matches = search_file(&params, path)?;
63
64        if params.simple_search {
65            // One-line occurrence count.
66            println!("{}: {} Occurrences(s)", params.query, line_matches.len())
67        } else {
68            // Print each matching line.
69            for line_match in line_matches {
70                println!("{line_match}");
71            }
72        }
73    }
74
75    Ok(())
76}
